-- 并发控制测试：2PL + No-Wait 死锁预防
-- 数据准备
create table concurrency_test (id int, name char(8), score float);
insert into concurrency_test values (1, 'xiaohong', 90.0);
insert into concurrency_test values (2, 'xiaoming', 95.0);
insert into concurrency_test values (3, 'zhanghua', 88.5);

-- 脏读测试：
-- 事务1: begin; update concurrency_test set score = 100.0 where id = 2; abort;
-- 事务2: begin; select * from concurrency_test where id = 2; commit;
-- 操作序列: t1a t2a t1b t2b t1c t1d
-- 预期: 事务2读到score=95.0（事务1的修改已回滚）

-- 脏写测试：
-- 事务1: begin; update concurrency_test set score = 100.0 where id = 2; abort;
-- 事务2: begin; update concurrency_test set score = 80.0 where id = 2; commit;
-- 预期: 最终score=80.0（事务1回滚，事务2提交）

-- 不可重复读测试：
-- 事务1: begin; select * from concurrency_test where id = 2;
-- 事务2: begin; update concurrency_test set score = 100.0 where id = 2; commit;
-- 事务1: select * from concurrency_test where id = 2; commit;
-- 预期: 两次读到相同的score（2PL隔离）
